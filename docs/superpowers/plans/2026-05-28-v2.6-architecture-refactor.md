```python
# backend/db/utils.py
"""DB-layer constants and pure utilities — no DB access."""
import asyncio
import logging
from datetime import datetime, timezone
from urllib.parse import urlparse, urlunparse

import httpx

logger = logging.getLogger(__name__)

DB_PATH = "hygie.db"

# Queue status values
STATUS_PENDING = "pending"
STATUS_DELETED = "deleted"
STATUS_ERROR   = "error"

# HTTP timeouts (seconds)
TIMEOUT_SHORT  = httpx.Timeout(10.0)
TIMEOUT_MEDIUM = httpx.Timeout(30.0)
TIMEOUT_LONG   = httpx.Timeout(60.0)


def now_utc() -> datetime:
    return datetime.now(timezone.utc)


def sanitize_url(url: str) -> str:
    """Remove auth credentials from URL for logging."""
    try:
        p = urlparse(url)
        safe = p._replace(netloc=p.hostname or "", query="", fragment="")
        return urlunparse(safe)
    except Exception:
        return url


def parse_iso_dt(s: str | None) -> datetime | None:
    if not s:
        return None
    try:
        return datetime.fromisoformat(s.replace("Z", "+00:00"))
    except Exception:
        return None


async def http_retry(fn, retries: int = 3, delay: float = 1.0):
    """Retry an async callable up to `retries` times on httpx errors."""
    last_exc: Exception | None = None
    for attempt in range(retries):
        try:
            return await fn()
        except (httpx.TimeoutException, httpx.NetworkError) as e:
            last_exc = e
            if attempt < retries - 1:
                await asyncio.sleep(delay * (attempt + 1))
    raise last_exc
```

```python
# backend/db/encryption.py
"""Fernet encryption helpers for sensitive settings."""
import logging
import os
from typing import Optional

logger = logging.getLogger(__name__)

_ENC_PREFIX = b"enc:"

SENSITIVE_KEYS = {
    "emby_api_key", "radarr_api_key", "sonarr_api_key",
    "seerr_api_key", "discord_webhook_url", "discord_webhook_url_2",
    "qbit_password", "admin_password_hash",
}


def _get_fernet():
    """Return a Fernet instance from HYGIE_ENCRYPTION_KEY env var, or None."""
    try:
        from cryptography.fernet import Fernet
        key = os.environ.get("HYGIE_ENCRYPTION_KEY", "").encode()
        if not key:
            return None
        return Fernet(key)
    except Exception as e:
        logger.warning(f"Fernet init failed: {e}")
        return None


def _encrypt_value(value: str, fernet=None) -> str:
    """Encrypt value if fernet available and value is non-empty plaintext."""
    if not value:
        return value
    f = fernet or _get_fernet()
    if not f:
        return value
    try:
        raw = value.encode()
        if raw.startswith(_ENC_PREFIX):
            return value  # already encrypted
        return (_ENC_PREFIX + f.encrypt(raw)).decode()
    except Exception as e:
        logger.warning(f"Encryption failed: {e}")
        return value


def _decrypt_value(value: str, fernet=None) -> str:
    """Decrypt value if it has the enc: prefix."""
    if not value:
        return value
    try:
        raw = value.encode()
        if not raw.startswith(_ENC_PREFIX):
            return value
        f = fernet or _get_fernet()
        if not f:
            return value
        return f.decrypt(raw[len(_ENC_PREFIX):]).decode()
    except Exception as e:
        logger.warning(f"Decryption failed: {e}")
        return value


async def _migrate_encrypt_settings(db) -> None:
    """Encrypt any plaintext sensitive settings on first run with encryption key."""
    f = _get_fernet()
    if not f:
        return
    async with db.execute("SELECT key, value FROM settings") as cur:
        rows = await cur.fetchall()
    for key, value in rows:
        if key in SENSITIVE_KEYS and value and not value.encode().startswith(_ENC_PREFIX):
            encrypted = _encrypt_value(value, f)
            await db.execute("UPDATE settings SET value=? WHERE key=?", (encrypted, key))
    await db.commit()
```

```python
# backend/db/websocket.py
"""In-process WebSocket broadcast to connected frontend clients."""
import asyncio
import logging
from typing import Any

logger = logging.getLogger(__name__)

_ws_clients: set = set()


def register_ws(ws) -> None:
    _ws_clients.add(ws)


def unregister_ws(ws) -> None:
    _ws_clients.discard(ws)


async def _broadcast(payload: dict[str, Any]) -> None:
    """Send JSON payload to all connected WS clients. Disconnected clients are silently removed."""
    dead = set()
    for ws in list(_ws_clients):
        try:
            await asyncio.wait_for(ws.send_json(payload), timeout=5.0)
        except Exception:
            dead.add(ws)
    _ws_clients.difference_update(dead)
```

```python
# backend/db/logs.py
"""Structured logging to DB + job history tracking."""
import logging
from typing import Optional

import aiosqlite

from .utils import DB_PATH, now_utc
from .settings_store import get_setting
from .websocket import _broadcast

logger = logging.getLogger(__name__)

_LOG_LEVELS = {"DEBUG": 10, "INFO": 20, "WARN": 30, "WARNING": 30, "ERROR": 40}


async def add_log(level: str, message: str, source: str = "system") -> None:
    configured = (await get_setting("log_level") or "INFO").upper()
    if _LOG_LEVELS.get(level.upper(), 0) < _LOG_LEVELS.get(configured, 20):
        return
    ts = now_utc().isoformat()
    try:
        async with aiosqlite.connect(DB_PATH) as db:
            await db.execute(
                "INSERT INTO logs (ts, level, message, source) VALUES (?, ?, ?, ?)",
                (ts, level.upper(), message, source),
            )
            await db.commit()
    except Exception as e:
        logger.error(f"add_log failed: {e}")
    await _broadcast({"type": "log", "ts": ts, "level": level, "message": message, "source": source})


async def add_job_run(job_type: str) -> int:
    ts = now_utc().isoformat()
    async with aiosqlite.connect(DB_PATH) as db:
        cursor = await db.execute(
            "INSERT INTO job_history (job_type, status, started_at) VALUES (?, 'running', ?)",
            (job_type, ts),
        )
        await db.commit()
        return cursor.lastrowid


async def finish_job_run(run_id: int, status: str, message: str = "") -> None:
    ts = now_utc().isoformat()
    async with aiosqlite.connect(DB_PATH) as db:
        await db.execute(
            "UPDATE job_history SET status=?, message=?, finished_at=? WHERE id=?",
            (status, message, ts, run_id),
        )
        await db.commit()
```

```python
# backend/proxy.py
"""Image proxy endpoint with SSRF-protection whitelist."""
import asyncio
import logging
import time
from urllib.parse import unquote, urlparse

import httpx
from fastapi import APIRouter, Request
from fastapi.responses import Response

from .database import get_setting, get_media_servers, sanitize_url

logger = logging.getLogger(__name__)

router = APIRouter()

_proxy_whitelist: set = set()
_proxy_whitelist_ts: float = 0.0
_proxy_whitelist_lock = asyncio.Lock()
_PROXY_WHITELIST_TTL = 300
_PROXY_MAX_BYTES = 10 * 1024 * 1024  # 10 MB


def invalidate_proxy_whitelist() -> None:
    global _proxy_whitelist_ts
    _proxy_whitelist_ts = 0.0


async def _get_proxy_whitelist() -> set:
    global _proxy_whitelist, _proxy_whitelist_ts
    if _proxy_whitelist and time.time() - _proxy_whitelist_ts < _PROXY_WHITELIST_TTL:
        return _proxy_whitelist
    async with _proxy_whitelist_lock:
        if _proxy_whitelist and time.time() - _proxy_whitelist_ts < _PROXY_WHITELIST_TTL:
            return _proxy_whitelist
        allowed = {
            "image.tmdb.org", "artworks.thetvdb.com", "thetvdb.com",
            "fanart.tv", "assets.fanart.tv",
        }
        for setting_key in ("radarr_url", "sonarr_url"):
            s = await get_setting(setting_key)
            if s:
                try:
                    h = (urlparse(s).hostname or "").lower()
                    if h:
                        allowed.add(h)
                except Exception:
                    pass
        for srv in await get_media_servers():
            for field in ("url", "ext_url"):
                u = (srv.get(field) or "").strip()
                if u:
                    try:
                        h = (urlparse(u).hostname or "").lower()
                        if h:
                            allowed.add(h)
                    except Exception:
                        pass
        _proxy_whitelist = allowed
        _proxy_whitelist_ts = time.time()
        return allowed


@router.get("/api/proxy/image")
async def proxy_image(request: Request):
    """Proxy images from configured services. SSRF-protected by host whitelist."""
    raw = request.url.query
    if not raw.startswith("url="):
        return Response(status_code=400)

    target_url = unquote(raw[4:])
    if not target_url:
        return Response(status_code=400)

    try:
        parsed = urlparse(target_url)
        if parsed.scheme not in ("http", "https"):
            return Response(status_code=400)

        host = (parsed.hostname or "").lower()
        allowed = await _get_proxy_whitelist()

        if host not in allowed:
            logger.warning(f"Proxy: host {host!r} not in whitelist")
            return Response(status_code=403, content=f"Host not allowed: {host}")

        async with httpx.AsyncClient(timeout=15, follow_redirects=True) as client:
            async with client.stream("GET", target_url) as r:
                if r.status_code == 200:
                    ct = r.headers.get("content-type", "image/jpeg")
                    if not ct.startswith("image/"):
                        return Response(status_code=415)
                    chunks: list[bytes] = []
                    total = 0
                    async for chunk in r.aiter_bytes(65536):
                        total += len(chunk)
                        if total > _PROXY_MAX_BYTES:
                            logger.warning(f"Proxy: response too large for {sanitize_url(target_url)[:80]}")
                            return Response(status_code=413)
                        chunks.append(chunk)
                    return Response(
                        content=b"".join(chunks),
                        media_type=ct,
                        headers={"Cache-Control": "public, max-age=3600"},
                    )
                if r.status_code != 500:
                    logger.warning(f"Proxy: upstream HTTP {r.status_code} for {sanitize_url(target_url)[:80]}")
    except Exception as e:
        logger.error(f"Proxy error: {e}")
    return Response(status_code=404)
```

```python
#!/usr/bin/env python3
"""i18n lint: report t() calls not in en.json, and en.json keys unused in JS/HTML."""
import json
import re
import sys
from pathlib import Path

ROOT = Path(__file__).parent.parent
I18N_FILE = ROOT / "frontend/static/i18n/en.json"
JS_DIRS = [ROOT / "frontend/static/js"]
HTML_DIRS = [ROOT / "frontend/templates"]


def load_keys() -> set[str]:
    with I18N_FILE.open() as f:
        return set(json.load(f).keys())


def find_t_calls() -> set[str]:
    """Find all t('...') and t("...") string literals in JS and HTML files."""
    pattern = re.compile(r"""t\(\s*['"](.+?)['"]\s*\)""")
    found = set()
    for d in JS_DIRS + HTML_DIRS:
        for f in d.rglob("*.js"):
            found.update(pattern.findall(f.read_text()))
        for f in d.rglob("*.html"):
            found.update(pattern.findall(f.read_text()))
    return found


def find_data_i18n() -> set[str]:
    """Find all data-i18n="..." attribute values."""
    pattern = re.compile(r"""data-i18n=["'](.+?)["']""")
    found = set()
    for d in HTML_DIRS:
        for f in d.rglob("*.html"):
            found.update(pattern.findall(f.read_text()))
    return found


def main() -> int:
    keys = load_keys()
    t_calls = find_t_calls()
    data_i18n = find_data_i18n()
    all_used = t_calls | data_i18n

    missing = sorted(all_used - keys)
    unused = sorted(keys - all_used)

    rc = 0
    if missing:
        print(f"\n❌ {len(missing)} key(s) used in JS/HTML but NOT in en.json:")
        for k in missing:
            print(f"   {k!r}")
        rc = 1
    else:
        print("✅ All t() calls have a matching key in en.json")

    if unused:
        print(f"\n⚠️  {len(unused)} key(s) in en.json NOT used in JS/HTML:")
        for k in unused:
            print(f"   {k!r}")
    else:
        print("✅ All en.json keys are used")

    return rc


if __name__ == "__main__":
    sys.exit(main())
```
